# Программная реализация

## Вычисление сил (потенциал Леннарда-Джонса)

```julia
@inbounds function compute_forces_lj!(positions, box, rc, ε, σ)
    N = size(positions, 1)
    forces = zeros(SVector{3,Float64}, N)
    energy = 0.0
    rc2 = rc^2
    
    for i in 1:N-1
        @simd for j in i+1:N
            dr = positions[i] - positions[j]
            dr -= round.(dr ./ box) .* box  # минимальное изображение
            r2 = dr·dr
            if r2 < rc2
                r2i = σ^2 / r2
                r6i = r2i^3
                r12i = r6i^2
                fij = 24ε * (2r12i - r6i) * r2i / r2
                forces[i] += fij * dr
                forces[j] -= fij * dr
                energy += 4ε * (r12i - r6i)
            end
        end
    end
    return forces, energy
end
```


## Интегратор Velocity-Verlet

```julia
function velocity_verlet!(pos, vel, forces, mass, dt, box, params)
    # Половина шага по скорости
    vel .+= 0.5 * dt * forces ./ mass
    
    # Обновление координат
    pos .+= dt * vel
    pos .= mod.(pos, box)  # периодические границы
    
    # Пересчёт сил
    forces_new, energy = compute_forces_lj!(pos, box, params.rc, params.ε, params.σ)
    
    # Вторая половина шага по скорости
    vel .+= 0.5 * dt * forces_new ./ mass
    
    return forces_new, energy
end
```


## Список соседей Верле (Verlet list)

```julia
function build_verlet_list(positions, box, rc, skin=0.3)
    rlist = rc + skin  # радиус списка с запасом
    rlist2 = rlist^2
    neighbor_list = [Int[] for _ in 1:size(positions,1)]
    
    for i in 1:N-1
        for j in i+1:N
            dr = positions[i] - positions[j]
            dr -= round.(dr ./ box) .* box
            if dr·dr < rlist2
                push!(neighbor_list[i], j)
                push!(neighbor_list[j], i)
            end
        end
    end
    return neighbor_list
end
```

### Пространственная декомпозиция
- Область моделирования разбивается на подобласти (3D-решётка)
- Каждый MPI-процесс отвечает за свою подобласть
- Обмен граничными частицами — неблокирующие коммуникации

```python
import plotly.graph_objects as go

def animate_trajectory(trajectory, step=5):
    frames = []
    for t in range(0, trajectory.shape[0], step):
        frame = go.Frame(
            data=[go.Scatter3d(
                x=trajectory[t,:,0],
                y=trajectory[t,:,1],
                z=trajectory[t,:,2],
                mode='markers',
                marker=dict(size=2, color='royalblue')
            )],
            name=f'frame_{t}'
        )
        frames.append(frame)
    
    fig = go.Figure(frames=frames)
    fig.update_layout(scene=dict(aspectmode='cube'))
    return fig
```


## Радиальная функция распределения g(r)

```python
def compute_rdf(positions, box, nbins=100, rc=2.5):
    hist = np.zeros(nbins)
    dr = rc / nbins
    N = len(positions)
    rho = N / (box[0] * box[1] * box[2])
    
    for i in range(N):
        for j in range(i+1, N):
            dr_vec = positions[i] - positions[j]
            dr_vec -= np.round(dr_vec / box) * box
            r = np.linalg.norm(dr_vec)
            if r < rc:
                bin_idx = int(r / dr)
                hist[bin_idx] += 2
    
    r_vals = np.linspace(0, rc, nbins)
    g_r = np.zeros(nbins)
    for i, r in enumerate(r_vals):
        if r > 0:
            volume = 4/3 * np.pi * ((r+dr)**3 - r**3)
            g_r[i] = hist[i] / (N * rho * volume)
    return r_vals, g_r
```


## Типовые графики анализа

| График | Формула | Назначение |
|--------|---------|------------|
| **Температурный профиль** | $T(t) = \frac{2}{3Nk_B} \sum \frac{p_i^2}{2m}$ | Контроль термостатирования |
| **Энергетический баланс** | $E_{\text{полн}} = E_k + E_p$ | Проверка сохранения энергии |
| **Среднеквадратичное смещение** | $\text{MSD}(t) = \langle |\vec{r}_i(t) - \vec{r}_i(0)|^2 \rangle$ | Коэффициент диффузии |
| **Радиальная функция** | $g(r) = \frac{V}{N^2} \langle \sum_{i \neq j} \delta(r - r_{ij}) \rangle$ | Идентификация фазового состояния |
