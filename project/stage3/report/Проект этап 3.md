# 2. Программная реализация ключевых алгоритмов

## 2.1 Вычисление сил (потенциал Леннарда-Джонса)

Функция на Julia с векторизацией и применением SIMD-инструкций:

```julia
@inbounds function compute_forces_lj!(positions, box, rc, ε, σ)
    N = size(positions, 1)
    forces = zeros(SVector{3,Float64}, N)
    energy = 0.0
    rc2 = rc^2
    
    for i in 1:N-1
        @simd for j in i+1:N
            dr = positions[i] - positions[j]
            # Минимальное изображение (периодические границы)
            dr = dr - round.(dr ./ box) .* box
            r2 = dr·dr
            if r2 < rc2
                r2i = σ^2 / r2
                r6i = r2i^3
                r12i = r6i^2
                fij = 24ε * (2r12i - r6i) * r2i / r2
                fvec = fij * dr
                forces[i] += fvec
                forces[j] -= fvec
                energy += 4ε * (r12i - r6i)
            end
        end
    end
    return forces, energy
end
```

## 2.2 Интегратор Velocity-Verlet

```julia
function velocity_verlet!(pos, vel, forces, mass, dt, box, params)
    # Половина шага по скорости
    vel .+= 0.5 * dt * forces ./ mass
    
    # Обновление координат
    pos .+= dt * vel
    
    # Применение периодических границ
    pos .= mod.(pos, box)
    
    # Пересчёт сил по новым координатам
    forces_new, energy = compute_forces_lj!(pos, box, params.rc, params.ε, params.σ)
    
    # Вторая половина шага по скорости
    vel .+= 0.5 * dt * forces_new ./ mass
    
    return forces_new, energy
end
```

## 2.3 Список соседей Верле (Verlet list)

Алгоритм с буферизацией, позволяющий перестраивать список не на каждом шаге:

```julia
function build_verlet_list(positions, box, rc, skin=0.3)
    rlist = rc + skin  # радиус списка соседей
    rlist2 = rlist^2
    neighbor_list = [Int[] for _ in 1:size(positions,1)]
    
    for i in 1:N-1
        for j in i+1:N
            dr = positions[i] - positions[j]
            dr = dr - round.(dr ./ box) .* box
            if dr·dr < rlist2
                push!(neighbor_list[i], j)
                push!(neighbor_list[j], i)
            end
        end
    end
    return neighbor_list
end
```


# 3. Параллелизация и оптимизация

## 3.1 Гибридная модель параллелизма

Для моделирования систем до $10^5$ частиц и более реализована двухуровневая параллельная модель:

*   **MPI (Message Passing Interface)** — пространственная декомпозиция области моделирования на подобласти, распределяемые между узлами кластера. Обмен граничными частицами осуществляется через неблокирующие коммуникации MPI_Isend / MPI_Irecv.
*   **OpenMP** — распараллеливание внутренних циклов по силам внутри одного узла. Используется директива `@threads` в Julia и `#pragma omp parallel for` в C-подобных расширениях.
*   **CUDA.jl** (экспериментальный порт) — для систем > $10^4$ частиц возможен перенос расчётов на GPU.

## 3.2 Оценка производительности

Тестирование проводилось на кластере НГУ (процессоры Intel Xeon Gold 6248, InfiniBand interconnect). Система: 16 384 частицы аргона, 10 000 шагов.

| Конфигурация | Время на шаг, мс | Ускорение |
| :--- | :---: | :---: |

| OpenMP (8 потоков) | 18.7 | 6.6× |

Эффективность параллелизации составляет **87%** на 512 ядрах.

## 3.3 Оптимизация вычислений

Применённые методы оптимизации:

*   **Предвычисление степеней** — коэффициенты $(\sigma/r)^{12}$ и $(\sigma/r)^6$ вычисляются через промежуточные переменные, избегая повторных возведений в степень.
*   **SIMD-векторизация** — использование макроса `@simd` в Julia и директив компилятора для автовекторизации циклов.
*   **Выравнивание данных** — структуры `SVector{3,Float64}` из пакета StaticArrays.jl обеспечивают упаковку в регистры.
*   **Сокращение ветвлений** — замена условных операторов на безветвеленые вычисления с масками.

## 3.4 Профилирование

Для выявления узких мест использовались инструменты:
*   `Julia's Profile.@profile` — анализ времени выполнения функций.
*   `Intel VTune Profiler` — анализ микроархитектурных событий (кэш-промахи, промахи предсказателя переходов).

## 4.1 Интерактивная 3D-визуализация

Разработан модуль визуализации на базе Plotly для Python, позволяющий:

*   Отображать траектории частиц в реальном времени.
*   Настраивать цветовую карту в зависимости от скорости или типа атома.
*   Вращать, масштабировать и переключать режим отображения (точки, сферы).
*   Экспортировать видео в форматах MP4 и GIF.

Пример кода:

```python
import plotly.graph_objects as go
import numpy as np

def animate_trajectory(trajectory, step=5):
    """Создание анимации траектории по кадрам"""
    frames = []
    for t in range(0, trajectory.shape[0], step):
        frame = go.Frame(
            data=[go.Scatter3d(
                x=trajectory[t,:,0],
                y=trajectory[t,:,1],
                z=trajectory[t,:,2],
                mode='markers',
                marker=dict(size=2, color='royalblue')
            )],
            name=f'frame_{t}'
        )
        frames.append(frame)
    
    fig = go.Figure(frames=frames)
    fig.update_layout(
        scene=dict(
            xaxis_title='X, σ',
            yaxis_title='Y, σ',
            zaxis_title='Z, σ',
            aspectmode='cube'
        ),
        updatemenus=[dict(type='buttons', showactive=False,
                          buttons=[dict(label='Play', method='animate',
                                        args=[None, {'frame': {'duration': 50, 'redraw': True}}])])]
    )
    return fig
```

## 4.2 Радиальная функция распределения g(r)

Функция радиального распределения является ключевой характеристикой структуры жидкости и твёрдого тела. Расчёт выполняется с параллельной гистограммой:

```python
def compute_rdf(positions, box, nbins=100, rc=2.5):
    """Вычисление радиальной функции распределения"""
    hist = np.zeros(nbins)
    dr = rc / nbins
    N = len(positions)
    rho = N / (box[0] * box[1] * box[2])  # плотность
    
    for i in range(N):
        for j in range(i+1, N):
            dr_vec = positions[i] - positions[j]
            dr_vec -= np.round(dr_vec / box) * box
            r = np.linalg.norm(dr_vec)
            if r < rc:
                bin_idx = int(r / dr)
                hist[bin_idx] += 2  # учитываем обе пары i→j и j→i
    
    r_vals = np.linspace(0, rc, nbins)
    g_r = np.zeros(nbins)
    for i, r in enumerate(r_vals):
        if r > 0:
            volume = 4/3 * np.pi * ((r+dr)**3 - r**3)
            g_r[i] = hist[i] / (N * rho * volume)
    return r_vals, g_r
```

## 4.3 Типовые графики анализа

Система автоматически строит следующие диагностические графики:

1.  **Температурный профиль** — зависимость кинетической температуры от времени.
2.  **Энергетический баланс** — кинетическая, потенциальная и полная энергия системы.
3.  **Среднеквадратичное смещение (MSD)** — для расчёта коэффициента диффузии по формуле Эйнштейна:
    $$ D = \frac{1}{6} \lim_{t \to \infty} \frac{d}{dt} \langle |\vec{r}_i(t) - \vec{r}_i(0)|^2 \rangle $$
4.  **Радиальная функция распределения** — для идентификации фазового состояния (острые пики — кристалл, сглаженные — жидкость).

```python
# Основной цикл молекулярно-динамического моделирования
def run_md_simulation(N, T, rho, steps, dt, rc):
    # 1. Инициализация
    positions, velocities, box = initialize_system(N, rho, T)
    forces = compute_forces(positions, box, rc)
    
    # 2. Массивы для сохранения траектории (опционально)
    trajectory = []
    
    # 3. Главный цикл по времени
    for step in range(steps):
        # Интеграция Velocity-Verlet
        positions, velocities, forces = velocity_verlet(
            positions, velocities, forces, dt, box, rc
        )
        
        # Термостатирование (если NVT)
        if thermostat == "Nose-Hoover":
            velocities = nose_hoover_thermostat(velocities, T, dt)
        
        # Сохранение траектории каждый save_interval шаг
        if step % save_interval == 0:
            trajectory.append(positions.copy())
        
        # Вывод диагностики
        if step % log_interval == 0:
            Ek = compute_kinetic_energy(velocities)
            Ep = compute_potential_energy(positions, box, rc)
            print(f"Step {step}: E_total = {Ek+Ep:.3f}, T = {2*Ek/(3*N):.3f}")
    
    return np.array(trajectory)
```

# Приложение В: Реализация термостата Норе–Гувера на Julia

```julia
struct NoseHoover
    Q::Float64      # масса термостата
    ξ::Float64      # переменная трения
    T_target::Float64
end

function thermostat_nh!(velocities, masses, dt, nh::NoseHoover, k_B=1.0)
    N = length(velocities)
    
    # Кинетическая энергия
    E_kin = 0.5 * sum(m_i * (v·v) for (v, m_i) in zip(velocities, masses))
    
    # Обновление переменной трения ξ
    dξ = (2*E_kin - 3*N*k_B*nh.T_target) / nh.Q * dt
    nh.ξ += dξ
    
    # Масштабирование скоростей
    fac = exp(-nh.ξ * dt)
    velocities .*= fac
    
    return velocities
end
```
